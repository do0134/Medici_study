1. N = int(input())
   crane = list(map(int, input().split()))
   M = int(input())
   box = list(map(int, input().split()))
   box.sort()
   n = 1
   flag = 0
   i = 0
   while sum(box):
       for j in range(-1, -M-1, -1):
           if i != N-1 and box[-1] <= crane[-1]:
               if box[j] != 0 and box[j] <= crane[i]:
                   box[j] = 0
                   i += 1
                   break

           elif i == N-1 and box[-1] <= crane[-1]:
               if box[j] != 0 and box[j] <= crane[i]:
                   box[j] = 0
                   n += 1
                   flag = n
                   i = 0
           else:
               flag = -1

   print(flag)
