## 다단계 칫솔 판매



```python
import math

def solution(enroll, referral, seller, amount):
    answer = []
    # dictionary 초기화
    my_dict = {'-': [0, 0]}
    N = len(enroll)
    for i in range(N):
        my_dict.setdefault(enroll[i], [referral[i], 0])
    
    # seller 순회
    M = len(seller)
    for i in range(M):
        k = seller[i]
        summ = amount[i] * 100
        my_dict[k][1] += math.ceil(summ * 0.9)
        summ = int(summ * 0.1)
        # center이거나 더이상 남겨줄 것이 없을 때 while 종료
        while k != '-' and summ > 0:
            k = my_dict[k][0]
            my_dict[k][1] += math.ceil(summ * 0.9)
            summ = int(summ * 0.1)
    
    for i in enroll:
        answer.append(my_dict[i][1])
    
        
    return answer
```
