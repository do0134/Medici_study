```python
from collections import deque

def solution(board):
    N = len(board)
    ei, ej = N-1, N-1
    def bfs(si, sj):
        q = deque()
        # 시작 좌표, 시작 방향으로 초기화
        q.append([si, sj, 0, 0])
        # 방문배열 초기화 [방문여부, 가격] (방문여부 체크할필요없을듯??)
        visited = [[[0, 99999999]] * N for _ in range(N)]
        visited[si][sj] = [1, -500]

        while q:
            #현재 좌표, 현재 방향
            ci, cj, cdi, cdj = q.popleft()

            for di, dj in [[0, 1], [0, -1], [1, 0], [-1, 0]]:
                ni, nj = ci + di, cj + dj
                if 0 <= ni < N and 0 <= nj < N and not board[ni][nj]:
                    # 같은방향
                    if cdi == di and cdj == dj:
                        price = 100
                    # 다른 방향
                    else:
                        price = 600
                    # 현재 배열 + PRICE 보다 다음 배열의 가격이 더 큰경우
                    if visited[ci][cj][1] + price <= visited[ni][nj][1]:
                        visited[ni][nj] = [1, visited[ci][cj][1] + price]
                        q.append([ni, nj, di, dj])

        return visited[ei][ej][1]

    answer = bfs(0, 0)
    return answer

```
