```
from collections import deque

N = int(input())
fork = sorted(list(map(int, input().split())), reverse=True)
M = int(input())
boxes = deque(sorted(list(map(int, input().split())), reverse=True))
cnt = 0

if boxes[0] > fork[0]:
    print(-1)
else:
    while boxes:
        cnt += 1
        for i in fork:
            if not boxes:
                break
                # 시간 초과 방지
            if boxes[-1] > i:
                continue
            if i >= boxes[0]:
                boxes.popleft()
            else:
                for j in range(len(boxes)):
                    if boxes[j] <= i:
                        del boxes[j]
                        break
    print(cnt)
```
